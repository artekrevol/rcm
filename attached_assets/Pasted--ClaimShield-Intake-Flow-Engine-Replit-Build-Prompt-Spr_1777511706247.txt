```sql
-- Flow definitions (the configurable sequence)
CREATE TABLE IF NOT EXISTS flows (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  description TEXT,
  trigger_event TEXT NOT NULL,        -- 'lead_created' | 'status_changed' | 'manual' | 'scheduled'
  trigger_conditions JSONB DEFAULT '{}',
  is_active BOOLEAN DEFAULT true,
  organization_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_flows_trigger_active ON flows(trigger_event, is_active);

-- Steps within a flow (ordered)
CREATE TABLE IF NOT EXISTS flow_steps (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  flow_id UUID NOT NULL REFERENCES flows(id) ON DELETE CASCADE,
  step_order INT NOT NULL,
  step_type TEXT NOT NULL,            -- 'call' | 'sms' | 'email' | 'wait' | 'branch' | 'vob_check' | 'human_task' | 'webhook'
  channel TEXT,                       -- 'vapi' | 'retell' | 'twilio' | 'gmail' | 'system' | NULL
  template_id UUID,                   -- references email_templates or sms_templates or call_scripts
  template_inline TEXT,               -- inline body if no template_id (for quick demos)
  delay_minutes INT DEFAULT 0,
  max_attempts INT DEFAULT 1,
  retry_delay_minutes INT DEFAULT 60,
  success_condition JSONB,            -- e.g. {"field": "vob_score", "op": ">=", "value": 80}
  on_success_step_id UUID,            -- self-FK to flow_steps
  on_failure_step_id UUID,
  on_response_step_id UUID,           -- if lead replies on any channel
  metadata JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_flow_steps_flow_order ON flow_steps(flow_id, step_order);

-- Each lead's progress through a flow
CREATE TABLE IF NOT EXISTS flow_runs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  flow_id UUID NOT NULL REFERENCES flows(id),
  lead_id INT NOT NULL,               -- match the existing leads.id type
  current_step_id UUID REFERENCES flow_steps(id),
  status TEXT NOT NULL DEFAULT 'running',  -- 'running' | 'paused' | 'completed' | 'failed' | 'cancelled'
  started_at TIMESTAMPTZ DEFAULT NOW(),
  next_action_at TIMESTAMPTZ DEFAULT NOW(),
  attempt_count INT DEFAULT 0,
  context JSONB DEFAULT '{}',         -- accumulated data (extracted insurance, etc.)
  completed_at TIMESTAMPTZ,
  failed_reason TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_flow_runs_status_action ON flow_runs(status, next_action_at) WHERE status = 'running';
CREATE INDEX IF NOT EXISTS idx_flow_runs_lead ON flow_runs(lead_id);

-- Audit trail of every step execution
CREATE TABLE IF NOT EXISTS flow_run_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  flow_run_id UUID NOT NULL REFERENCES flow_runs(id) ON DELETE CASCADE,
  step_id UUID REFERENCES flow_steps(id),
  event_type TEXT NOT NULL,           -- 'step_started' | 'step_completed' | 'step_failed' | 'response_received' | 'lock_acquired' | 'lock_released'
  channel TEXT,
  payload JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_flow_run_events_run ON flow_run_events(flow_run_id, created_at);

-- The cross-channel lock (prevents double engagement)
CREATE TABLE IF NOT EXISTS comm_locks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id INT NOT NULL,
  acquired_by_type TEXT NOT NULL,     -- 'flow_run' | 'manual_user' | 'inbound_response'
  acquired_by_id TEXT NOT NULL,       -- flow_run_id, user_id, or event_id as string
  channel TEXT NOT NULL,              -- 'call' | 'sms' | 'email' | 'any'
  reason TEXT,
  acquired_at TIMESTAMPTZ DEFAULT NOW(),
  expires_at TIMESTAMPTZ NOT NULL,
  released_at TIMESTAMPTZ
);

CREATE INDEX IF NOT EXISTS idx_comm_locks_active ON comm_locks(lead_id, released_at, expires_at) WHERE released_at IS NULL;
```

## Item 2 — The orchestrator (server-side cron)

Create `server/jobs/flow-orchestrator.ts`. This is the heart of the engine. It runs every 60 seconds via `setInterval` (same pattern as the Stedi 277/835 polling — Railway runs the server continuously, no separate cron service needed).

```typescript
// server/jobs/flow-orchestrator.ts
import { db } from '../db';
import { acquireLock, releaseLock, hasActiveLock } from '../services/comm-locks';
import { executeStep } from '../services/flow-step-executor';
import { logFlowEvent } from '../services/flow-events';

const TICK_INTERVAL_MS = 60_000;
let isRunning = false;

export function startFlowOrchestrator() {
  console.log('[flow-orchestrator] starting, tick interval =', TICK_INTERVAL_MS, 'ms');
  setInterval(tick, TICK_INTERVAL_MS);
  // also tick once at startup
  tick();
}

async function tick() {
  if (isRunning) {
    console.log('[flow-orchestrator] previous tick still running, skipping');
    return;
  }
  isRunning = true;
  try {
    const dueRuns = await db.query(`
      SELECT fr.*, fs.id as step_id, fs.step_type, fs.channel, fs.max_attempts, fs.retry_delay_minutes
      FROM flow_runs fr
      LEFT JOIN flow_steps fs ON fs.id = fr.current_step_id
      WHERE fr.status = 'running'
        AND fr.next_action_at <= NOW()
      ORDER BY fr.next_action_at ASC
      LIMIT 50
    `);

    for (const run of dueRuns.rows) {
      // Check for active lock — skip and reschedule if locked
      const locked = await hasActiveLock(run.lead_id);
      if (locked) {
        await db.query(
          `UPDATE flow_runs SET next_action_at = NOW() + INTERVAL '15 minutes' WHERE id = $1`,
          [run.id]
        );
        await logFlowEvent(run.id, run.step_id, 'step_skipped_locked', {});
        continue;
      }

      try {
        await executeStep(run);
      } catch (err) {
        console.error('[flow-orchestrator] step execution failed:', err);
        await logFlowEvent(run.id, run.step_id, 'step_failed', { error: String(err) });
        // Mark run as failed (don't retry forever)
        await db.query(
          `UPDATE flow_runs SET status = 'failed', failed_reason = $2 WHERE id = $1`,
          [run.id, String(err).slice(0, 500)]
        );
      }
    }
  } finally {
    isRunning = false;
  }
}
```

Wire it into `server/index.ts` AFTER the express app starts:

## Item 3 — Comm lock service

Create `server/services/comm-locks.ts`:

```typescript
import { db } from '../db';

export async function hasActiveLock(leadId: number, channel?: string): Promise<boolean> {
  const result = await db.query(
    `SELECT id FROM comm_locks 
     WHERE lead_id = $1 
       AND released_at IS NULL 
       AND expires_at > NOW()
       AND ($2::text IS NULL OR channel = $2 OR channel = 'any')
     LIMIT 1`,
    [leadId, channel || null]
  );
  return result.rows.length > 0;
}

export async function acquireLock(opts: {
  leadId: number;
  acquiredByType: 'flow_run' | 'manual_user' | 'inbound_response';
  acquiredById: string;
  channel: 'call' | 'sms' | 'email' | 'any';
  reason: string;
  durationMinutes: number;
}): Promise<string | null> {
  // Double-check no conflicting lock exists
  const conflict = await hasActiveLock(opts.leadId, opts.channel);
  if (conflict) return null;

  const result = await db.query(
    `INSERT INTO comm_locks (lead_id, acquired_by_type, acquired_by_id, channel, reason, expires_at)
     VALUES ($1, $2, $3, $4, $5, NOW() + ($6 || ' minutes')::interval)
     RETURNING id`,
    [opts.leadId, opts.acquiredByType, opts.acquiredById, opts.channel, opts.reason, opts.durationMinutes]
  );
  return result.rows[0].id;
}

export async function releaseLock(lockId: string): Promise<void> {
  await db.query(
    `UPDATE comm_locks SET released_at = NOW() WHERE id = $1 AND released_at IS NULL`,
    [lockId]
  );
}

export async function releaseAllLocksForLead(leadId: number): Promise<void> {
  await db.query(
    `UPDATE comm_locks SET released_at = NOW() WHERE lead_id = $1 AND released_at IS NULL`,
    [leadId]
  );
}

// Default lock durations by channel
export const LOCK_DURATION_MINUTES: Record<string, number> = {
  call: 30,    // 30 min covers a long call + transcript processing
  sms: 5,      // 5 min covers Twilio delivery + immediate read
  email: 5,    // 5 min covers SMTP send
  any: 240,    // 4 hours — used when human needs to handle a response
};
```

## Item 4 — Step executor with channel dispatchers

Create `server/services/flow-step-executor.ts`. This is the dispatcher that maps `step_type` to the actual channel handler:

```typescript
import { db } from '../db';
import { acquireLock, releaseLock, LOCK_DURATION_MINUTES } from './comm-locks';
import { logFlowEvent } from './flow-events';
import { runStediEligibility } from './stedi-eligibility';  // reuse from billing
import { initiateVapiCall } from './vapi-client';            // existing
import { sendTwilioSms } from './twilio-client';             // existing
import { sendGmailEmail } from './gmail-client';             // existing

export async function executeStep(run: any): Promise<void> {
  const { id: runId, lead_id: leadId, step_id: stepId, step_type, channel } = run;
  
  await logFlowEvent(runId, stepId, 'step_started', { stepType: step_type });

  switch (step_type) {
    case 'wait':
      await advanceToNextStep(runId, 'success');
      break;

    case 'call':
      await handleCallStep(run);
      break;

    case 'sms':
      await handleSmsStep(run);
      break;

    case 'email':
      await handleEmailStep(run);
      break;

    case 'vob_check':
      await handleVobStep(run);
      break;

    case 'branch':
      await handleBranchStep(run);
      break;

    case 'human_task':
      // Pause the flow and notify
      await db.query(
        `UPDATE flow_runs SET status = 'paused', updated_at = NOW() WHERE id = $1`,
        [runId]
      );
      await logFlowEvent(runId, stepId, 'human_task_created', {});
      // TODO: push notification to assigned coordinator
      break;

    default:
      throw new Error(`Unknown step type: ${step_type}`);
  }
}

async function handleCallStep(run: any) {
  const lockId = await acquireLock({
    leadId: run.lead_id,
    acquiredByType: 'flow_run',
    acquiredById: run.id,
    channel: 'any',  // calls block ALL other channels
    reason: `flow step ${run.step_id} — outbound call`,
    durationMinutes: LOCK_DURATION_MINUTES.call,
  });

  if (!lockId) {
    // Reschedule
    await db.query(
      `UPDATE flow_runs SET next_action_at = NOW() + INTERVAL '15 minutes' WHERE id = $1`,
      [run.id]
    );
    return;
  }

  try {
    const lead = await db.query(`SELECT * FROM leads WHERE id = $1`, [run.lead_id]);
    if (!lead.rows[0]) throw new Error('Lead not found');

    // Get the script from template_inline or template_id
    const step = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [run.step_id]);
    const script = step.rows[0].template_inline || 'You are a Caritas Senior Care intake coordinator. Confirm the caller wants senior care services, capture insurance carrier, member ID, date of birth, and care urgency. Be warm and concise.';

    const callResult = await initiateVapiCall({
      leadId: run.lead_id,
      phone: lead.rows[0].phone,
      script,
      flowRunId: run.id,
      lockId,
    });

    // Vapi webhook will release the lock and advance the step when call ends
    await logFlowEvent(run.id, run.step_id, 'call_initiated', { vapiCallId: callResult.vapiCallId, lockId });
    
    // Don't advance step here — webhook will do it when call completes
    await db.query(
      `UPDATE flow_runs SET next_action_at = NOW() + INTERVAL '30 minutes', updated_at = NOW() WHERE id = $1`,
      [run.id]
    );
  } catch (err) {
    await releaseLock(lockId);
    throw err;
  }
}

async function handleSmsStep(run: any) {
  const lockId = await acquireLock({
    leadId: run.lead_id,
    acquiredByType: 'flow_run',
    acquiredById: run.id,
    channel: 'sms',
    reason: `flow step ${run.step_id} — outbound SMS`,
    durationMinutes: LOCK_DURATION_MINUTES.sms,
  });

  if (!lockId) {
    await db.query(
      `UPDATE flow_runs SET next_action_at = NOW() + INTERVAL '15 minutes' WHERE id = $1`,
      [run.id]
    );
    return;
  }

  try {
    const lead = await db.query(`SELECT * FROM leads WHERE id = $1`, [run.lead_id]);
    const step = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [run.step_id]);
    const body = renderTemplate(step.rows[0].template_inline || '', lead.rows[0]);

    await sendTwilioSms({ to: lead.rows[0].phone, body });
    await logFlowEvent(run.id, run.step_id, 'sms_sent', { body });

    await advanceToNextStep(run.id, 'success');
  } finally {
    await releaseLock(lockId);
  }
}

async function handleEmailStep(run: any) {
  const lockId = await acquireLock({
    leadId: run.lead_id,
    acquiredByType: 'flow_run',
    acquiredById: run.id,
    channel: 'email',
    reason: `flow step ${run.step_id} — outbound email`,
    durationMinutes: LOCK_DURATION_MINUTES.email,
  });

  if (!lockId) {
    await db.query(
      `UPDATE flow_runs SET next_action_at = NOW() + INTERVAL '15 minutes' WHERE id = $1`,
      [run.id]
    );
    return;
  }

  try {
    const lead = await db.query(`SELECT * FROM leads WHERE id = $1`, [run.lead_id]);
    const step = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [run.step_id]);
    const subject = renderTemplate(step.rows[0].metadata?.subject || 'Following up', lead.rows[0]);
    const body = renderTemplate(step.rows[0].template_inline || '', lead.rows[0]);

    await sendGmailEmail({ to: lead.rows[0].email, subject, body });
    await logFlowEvent(run.id, run.step_id, 'email_sent', { subject });

    await advanceToNextStep(run.id, 'success');
  } finally {
    await releaseLock(lockId);
  }
}

async function handleVobStep(run: any) {
  // No comm lock needed — this is a system call to Stedi
  const lead = await db.query(`SELECT * FROM leads WHERE id = $1`, [run.lead_id]);
  const l = lead.rows[0];

  if (!l.insurance_carrier || !l.insurance_member_id || !l.date_of_birth) {
    await logFlowEvent(run.id, run.step_id, 'vob_skipped_missing_data', {});
    await advanceToNextStep(run.id, 'failure');
    return;
  }

  try {
    const result = await runStediEligibility({
      payerId: mapCarrierToStediPayerId(l.insurance_carrier),
      memberId: l.insurance_member_id,
      dateOfBirth: l.date_of_birth,
      firstName: l.name?.split(' ')[0],
      lastName: l.name?.split(' ').slice(1).join(' '),
    });

    // Write to vob_verifications, update lead.vob_score
    await db.query(
      `INSERT INTO vob_verifications (lead_id, status, payer_name, copay, deductible, oop_max, network_status, created_at)
       VALUES ($1, $2, $3, $4, $5, $6, $7, NOW())`,
      [run.lead_id, 'verified', result.payerName, result.copay, result.deductible, result.oopMax, result.networkStatus]
    );

    await db.query(
      `UPDATE leads SET vob_score = 100, updated_at = NOW() WHERE id = $1`,
      [run.lead_id]
    );

    await logFlowEvent(run.id, run.step_id, 'vob_completed', result);
    await advanceToNextStep(run.id, 'success');
  } catch (err) {
    await logFlowEvent(run.id, run.step_id, 'vob_failed', { error: String(err) });
    await advanceToNextStep(run.id, 'failure');
  }
}

async function handleBranchStep(run: any) {
  // Evaluate success_condition against the run's context + lead state
  const step = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [run.step_id]);
  const lead = await db.query(`SELECT * FROM leads WHERE id = $1`, [run.lead_id]);
  const condition = step.rows[0].success_condition;

  const passed = evaluateCondition(condition, { ...lead.rows[0], ...run.context });
  await advanceToNextStep(run.id, passed ? 'success' : 'failure');
}

async function advanceToNextStep(runId: string, outcome: 'success' | 'failure' | 'response') {
  const run = await db.query(`SELECT * FROM flow_runs WHERE id = $1`, [runId]);
  const currentStep = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [run.rows[0].current_step_id]);
  const cs = currentStep.rows[0];

  let nextStepId: string | null = null;
  if (outcome === 'success') nextStepId = cs.on_success_step_id;
  else if (outcome === 'failure') nextStepId = cs.on_failure_step_id;
  else if (outcome === 'response') nextStepId = cs.on_response_step_id;

  // Default: next step in step_order
  if (!nextStepId) {
    const next = await db.query(
      `SELECT id FROM flow_steps WHERE flow_id = $1 AND step_order > $2 ORDER BY step_order ASC LIMIT 1`,
      [cs.flow_id, cs.step_order]
    );
    nextStepId = next.rows[0]?.id || null;
  }

  if (!nextStepId) {
    // Flow complete
    await db.query(
      `UPDATE flow_runs SET status = 'completed', completed_at = NOW(), updated_at = NOW() WHERE id = $1`,
      [runId]
    );
    await logFlowEvent(runId, cs.id, 'flow_completed', {});
    return;
  }

  const nextStep = await db.query(`SELECT * FROM flow_steps WHERE id = $1`, [nextStepId]);
  const delayMin = nextStep.rows[0].delay_minutes || 0;

  await db.query(
    `UPDATE flow_runs 
     SET current_step_id = $1, 
         next_action_at = NOW() + ($2 || ' minutes')::interval,
         attempt_count = 0,
         updated_at = NOW()
     WHERE id = $3`,
    [nextStepId, delayMin, runId]
  );

  await logFlowEvent(runId, nextStepId, 'step_scheduled', { delayMinutes: delayMin });
}

function renderTemplate(tmpl: string, lead: any): string {
  return tmpl
    .replace(/\{name\}/g, lead.name || '')
    .replace(/\{first_name\}/g, (lead.name || '').split(' ')[0])
    .replace(/\{phone\}/g, lead.phone || '')
    .replace(/\{state\}/g, lead.state || '');
}

function evaluateCondition(cond: any, data: any): boolean {
  if (!cond) return true;
  const { field, op, value } = cond;
  const actual = data[field];
  switch (op) {
    case '>=': return Number(actual) >= Number(value);
    case '<=': return Number(actual) <= Number(value);
    case '==': return actual == value;
    case '!=': return actual != value;
    case 'truthy': return !!actual;
    default: return false;
  }
}

function mapCarrierToStediPayerId(carrier: string): string {
  // Reuse the mapping from billing-side. Add new ones as needed.
  const map: Record<string, string> = {
    'BCBS': 'BCBSTX',
    'UHC': '87726',
    'Aetna': '60054',
    'Cigna': '62308',
    'Humana': '61101',
    'Medicare': '00973',
    'Medicare Advantage': '00973',
    'Medicaid': 'MCDTX',
  };
  return map[carrier] || carrier;
}
```

```typescript
import { db } from '../db';

export async function triggerMatchingFlows(opts: {
  event: string;
  leadId: number;
  context: Record<string, any>;
}) {
  // Find active flows whose trigger matches
  const flows = await db.query(
    `SELECT * FROM flows WHERE trigger_event = $1 AND is_active = true`,
    [opts.event]
  );

  for (const flow of flows.rows) {
    const conditions = flow.trigger_conditions || {};
    const matches = Object.entries(conditions).every(
      ([key, val]) => opts.context[key] === val
    );
    if (!matches) continue;

    // Find first step
    const firstStep = await db.query(
      `SELECT * FROM flow_steps WHERE flow_id = $1 ORDER BY step_order ASC LIMIT 1`,
      [flow.id]
    );
    if (!firstStep.rows[0]) continue;

    // Create flow_run
    const delayMin = firstStep.rows[0].delay_minutes || 0;
    await db.query(
      `INSERT INTO flow_runs (flow_id, lead_id, current_step_id, status, next_action_at, context)
       VALUES ($1, $2, $3, 'running', NOW() + ($4 || ' minutes')::interval, $5)`,
      [flow.id, opts.leadId, firstStep.rows[0].id, delayMin, JSON.stringify(opts.context)]
    );
  }
}
```

```typescript
export async function extractInsuranceFromTranscript(transcript: string) {
  const response = await anthropic.messages.create({
    model: 'claude-haiku-4-5-20251001',
    max_tokens: 500,
    messages: [{
      role: 'user',
      content: `Extract structured data from this intake call transcript. Return ONLY a JSON object with these keys: carrier (string), memberId (string), dob (YYYY-MM-DD), state (2-letter), urgency (immediate|within_week|within_month|exploring), serviceType (string), consent (boolean). If a field is unknown, use null. Do not include markdown or explanation, only JSON.

Transcript:
${transcript}`
    }]
  });
  
  const text = response.content.find(b => b.type === 'text')?.text || '{}';
  return JSON.parse(text.replace(/```json|```/g, '').trim());
}
```
