e) Verify with these queries against PRODUCTION_DATABASE_URL:

   -- All claims should have evaluation timestamps now
   SELECT COUNT(*) AS total,
          COUNT(*) FILTER (WHERE last_risk_evaluation_at IS NOT NULL) AS evaluated,
          COUNT(*) FILTER (WHERE last_risk_factors IS NOT NULL) AS with_factors
   FROM claims;
   
   -- Distribution of risk factor counts
   SELECT 
     CASE 
       WHEN jsonb_array_length(last_risk_factors) = 0 THEN 'clean'
       WHEN jsonb_array_length(last_risk_factors) BETWEEN 1 AND 2 THEN 'few'
       WHEN jsonb_array_length(last_risk_factors) BETWEEN 3 AND 5 THEN 'several'
       ELSE 'many'
     END AS factor_bucket,
     COUNT(*) 
   FROM claims 
   WHERE last_risk_factors IS NOT NULL
   GROUP BY factor_bucket;
   
   -- Sample 3 claims with their factors so I can spot-check
   SELECT id, status, jsonb_array_length(last_risk_factors) AS factors,
          last_risk_factors
   FROM claims
   WHERE last_risk_factors IS NOT NULL
   ORDER BY created_at DESC
   LIMIT 3;
