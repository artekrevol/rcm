        // Save connected status to practice settings

          `UPDATE practice_settings SET

            updated_at = NOW()`,

      res.json(result);
    } catch (err: any) {
      res.json({ success: false, message: err.message });
    }
  }
);
Add testOAConnection to server/services/office-ally.ts:
typescriptexport async function testOAConnection(
  username: string, 
  password: string
): Promise<{ success: boolean; message: string }> {
  const sftp = new Client();
  try {
    await sftp.connect({
      host: 'sftp.officeally.com',
      port: 22,
      username,
      password,
    });
    await sftp.end();
    return { success: true, message: 'Connected to Office Ally SFTP successfully' };
  } catch (err: any) {
    await sftp.end().catch(() => {});
    return { 
      success: false, 
      message: `Connection failed: ${err.message}. Check your Office Ally username and password.`
    };
  }
}

Add the state and handlers to the settings component:
typescriptconst [oaForm, setOaForm] = useState({
  submitterId: practiceSettings?.oa_submitter_id || '',
  username: practiceSettings?.oa_sftp_username || '',
  password: '',
});
const [testingConnection, setTestingConnection] = useState(false);
const [connectionTestResult, setConnectionTestResult] = useState<{
  success: boolean; 
  message: string;
} | null>(null);

const handleTestConnection = async () => {
  setTestingConnection(true);
  setConnectionTestResult(null);
  try {
    const res = await fetch('/api/billing/test-oa-connection', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        username: oaForm.username, 
        password: oaForm.password 
      }),
    });
    const result = await res.json();
    setConnectionTestResult(result);
  } catch {
    setConnectionTestResult({ 
      success: false, 
      message: 'Network error — could not reach the server' 
    });
  } finally {
    setTestingConnection(false);
  }
};

const handleSaveOASettings = async () => {
  await fetch('/api/billing/practice-settings', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      ...practiceSettings,
      oa_submitter_id: oaForm.submitterId,
      oa_sftp_username: oaForm.username,
      oa_sftp_password: oaForm.password,
    }),
  });
  toast({ title: 'Office Ally credentials saved' });
  queryClient.invalidateQueries(['practice-settings']);
};

Add the submit handler:
typescriptconst handleSubmitViaOfficeAlly = async () => {
  setSubmitting(true);
  try {
    const res = await fetch(`/api/billing/claims/${claim.id}/submit`, {
      method: 'POST',
    });
    const result = await res.json();
    
    if (result.success) {
      toast({ 
        title: 'Claim submitted successfully',
        description: `File: ${result.filename}` 
      });
      queryClient.invalidateQueries(['claim', claim.id]);
    } else {
      toast({ 
        title: 'Submission failed',
        description: result.error,
        variant: 'destructive'
      });
    }
  } finally {
    setSubmitting(false);
  }
};
Add the submit endpoint to routes.ts:
typescript// POST /api/billing/claims/:id/submit — submit via Office Ally SFTP
app.post('/api/billing/claims/:id/submit',
  requireRole('admin', 'rcm_manager'),
  async (req, res) => {
    try {
      const claim = await getClaimWithAllData(req.params.id);
      if (!claim) return res.status(404).json({ error: 'Claim not found' });
      
      const [practice, patient, provider, payer] = await Promise.all([
        getPracticeSettings(),
        getPatient(claim.patient_id),
        getProvider(claim.provider_id),
        getPayer(claim.payer),
      ]);
      
      const { submitClaim837P } = await import('./services/office-ally');
      const result = await submitClaim837P({ 
        claim, practice, patient, provider, payer 
      });
      
      if (result.success) {
        await pool.query(
          `UPDATE claims SET status = 'submitted', submitted_at = NOW() WHERE id = $1`,
          [claim.id]
        );
        
        await logActivity({
          activity_type: 'edi_submitted',
          claim_id: claim.id,
          performed_by: req.user.id,
          detail: `837P submitted via Office Ally: ${result.filename}`,
        });
      }
      
      res.json(result);
    } catch (err: any) {
      res.status(500).json({ success: false, error: err.message });
    }
  }
);
