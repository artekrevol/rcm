Fix 1 — Session table initialization (permanent fix)
In server/auth.ts, find the connect-pg-simple session store configuration. Remove createTableIfMissing: true entirely — this option reads a SQL file from the package directory at runtime which does not exist in production builds.
Replace it with inline session table creation that runs once when the server starts. Find the database initialization block in server/index.ts or wherever the pool is first used, and add this before any routes are registered:
typescript// Ensure session table exists — runs on every startup, safe to repeat
async function ensureSessionTable() {
  await pool.query(`
    CREATE TABLE IF NOT EXISTS "session" (
      "sid" varchar NOT NULL COLLATE "default",
      "sess" json NOT NULL,
      "expire" timestamp(6) NOT NULL,
      CONSTRAINT "session_pkey" PRIMARY KEY ("sid") NOT DEFERRABLE INITIALLY IMMEDIATE
    );
    CREATE INDEX IF NOT EXISTS "IDX_session_expire" ON "session" ("expire");
  `);
}
Call await ensureSessionTable() during server startup before app.listen(). The IF NOT EXISTS clauses make this completely safe to run on every deploy.

Fix 2 — Centralized user creation service (permanent fix)
The root cause of the plain text password problem is that user creation happens in multiple places (seed scripts, admin routes, future signup) and each place is responsible for remembering to hash the password. That's a bug waiting to happen every time.
Create a single user service file at server/services/user-service.ts that is the only place in the entire codebase that creates or updates users. Every other part of the system calls this service — nothing ever writes directly to the users table.
typescript// server/services/user-service.ts

import bcrypt from 'bcryptjs';
import { pool } from '../db';

const SALT_ROUNDS = 10;

export interface CreateUserParams {
  email: string;
  password: string;
  role: 'admin' | 'rcm_manager' | 'intake' | 'viewer';
  firstName?: string;
  lastName?: string;
}

export interface UpdatePasswordParams {
  userId: string;
  newPassword: string;
}

// The ONLY function that creates users anywhere in the system.
// Always hashes the password. Never stores plain text.
export async function createUser(params: CreateUserParams) {
  const { email, password, role, firstName, lastName } = params;

  // Validate before touching the database
  if (!email || !password || !role) {
    throw new Error('email, password, and role are required');
  }
  if (password.length < 8) {
    throw new Error('Password must be at least 8 characters');
  }
  
  const validRoles = ['admin', 'rcm_manager', 'intake', 'viewer'];
  if (!validRoles.includes(role)) {
    throw new Error(`Invalid role. Must be one of: ${validRoles.join(', ')}`);
  }

  // Check for duplicate email
  const existing = await pool.query(
    'SELECT id FROM users WHERE LOWER(email) = LOWER($1)',
    [email]
  );
  if (existing.rows.length > 0) {
    throw new Error(`A user with email ${email} already exists`);
  }

  // Hash always happens here — nowhere else
  const passwordHash = await bcrypt.hash(password, SALT_ROUNDS);

  const result = await pool.query(
    `INSERT INTO users (id, email, password, role, first_name, last_name, created_at, updated_at)
     VALUES (gen_random_uuid()::text, $1, $2, $3, $4, $5, NOW(), NOW())
     RETURNING id, email, role, first_name, last_name, created_at`,
    [email.toLowerCase(), passwordHash, role, firstName || null, lastName || null]
  );

  return result.rows[0];
}

// The ONLY function that updates passwords anywhere in the system.
export async function updatePassword(params: UpdatePasswordParams) {
  const { userId, newPassword } = params;

  if (!newPassword || newPassword.length < 8) {
    throw new Error('Password must be at least 8 characters');
  }

  const passwordHash = await bcrypt.hash(newPassword, SALT_ROUNDS);

  await pool.query(
    'UPDATE users SET password = $1, updated_at = NOW() WHERE id = $2',
    [passwordHash, userId]
  );
}

// Safe user lookup — never returns the password hash
export async function getUserByEmail(email: string) {
  const result = await pool.query(
    'SELECT id, email, role, first_name, last_name, created_at FROM users WHERE LOWER(email) = LOWER($1)',
    [email]
  );
  return result.rows[0] || null;
}

// For authentication only — returns the hash so Passport can compare
export async function getUserForAuth(email: string) {
  const result = await pool.query(
    'SELECT id, email, password, role FROM users WHERE LOWER(email) = LOWER($1)',
    [email]
  );
  return result.rows[0] || null;
}

Fix 3 — Update auth to use the service
In server/auth.ts, update the Passport local strategy to use getUserForAuth from the user service instead of querying the database directly:
typescriptimport { getUserForAuth } from './services/user-service';

passport.use(new LocalStrategy(
  { usernameField: 'email' },
  async (email, password, done) => {
    try {
      const user = await getUserForAuth(email);
      if (!user) {
        return done(null, false, { message: 'Invalid credentials' });
      }
      const isValid = await bcrypt.compare(password, user.password);
      if (!isValid) {
        return done(null, false, { message: 'Invalid credentials' });
      }
      return done(null, user);
    } catch (err) {
      return done(err);
    }
  }
));

Fix 5 — Fix the broken passwords in the Railway database right now
Add a one-time migration endpoint that re-hashes all users whose passwords are not bcrypt hashes. A bcrypt hash always starts with $2a$ or $2b$. Any password that doesn't start with that pattern is plain text and needs to be fixed.
typescript// Add to routes temporarily — remove after running once
app.post('/api/admin/rehash-passwords',
  requireRole('admin'),
  async (req, res) => {
    const users = await pool.query(
      "SELECT id, password FROM users WHERE password NOT LIKE '$2%'"
    );
    
    let fixed = 0;
    for (const user of users.rows) {
      const hash = await bcrypt.hash(user.password, 10);
      await pool.query(
        'UPDATE users SET password = $1 WHERE id = $2',
        [hash, user.id]
      );
      fixed++;
    }
    
    res.json({ 
      message: `Fixed ${fixed} user(s)`,
      affected: users.rows.length 
    });
  }
);

Backend endpoints:

app.get('/api/admin/users', requireRole('admin'), async (req, res) => {
